# test errors in test command
define (f x y z) = (+ (* x y) (- z 3))
define (g x) = (+ 0 x)

# check parentheses errors on the first argument
test f 1 2 3       4
test (f 1 2 3          6
test f 1 2 3)     8
test ((f 1 3 3)) 8 # multiple parentheses is regarded as errors here
test (f (g 0.56) g 7.8 6) 

# check parameter values errors on the first argument
test (f1 1 3 3) 4
test (f 1b 2 3) 4
test (f b1 2 j3) 4

# check errors on second argument (second argument can only be a number)
test (f 1 2 3) 46.7abs
test (f 1 2 3) a456.5
test (f 1 2 3) (46.7) # parentheses on second argument are regarded as error here
